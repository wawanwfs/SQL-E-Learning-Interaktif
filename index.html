query.sql
SELECT p.kota,
       COUNT(ps.pesanan_id) AS order_count,
       SUM(ps.total) AS omzet
FROM pesanan ps
JOIN pelanggan p ON p.pelanggan_id = ps.pelanggan_id
GROUP BY p.kota
ORDER BY omzet DESC;
